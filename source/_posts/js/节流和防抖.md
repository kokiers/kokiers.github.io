```bash
function debounce(func, wait=500){
    let timeout;
    return function(event){
      clearTimeout(timeout)
      timeout = setTimeout(()=>{
        func.call(this, event)
      },wait)
    }
}
```

```bash
function throttle(fn, delay=1000) {
    let flag = true;
    return function () {
        if (!flag) { return; }
        flag = false;
        setTimeout(() => {
            fn.apply(this, arguments);
            flag = true;
        }, delay);
    }
}
```
