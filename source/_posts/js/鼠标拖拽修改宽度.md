```bash
//html：
 <div class="ms" id="drags" :style="{'height': wrapHei + 'px','cursor':cursor}">
    <div class="ms-menu" :style="{'width':menuWidth + 'px'}">
    </div>
    <div class="ms-cont">
    </div>
</div>

window.addEventListener('mousemove', handleMove)
window.removeEventListener('mousemove',handleMove)

function handleMove(e){
  let menu = this.$refs.drags.getBoundingClientRect()
  let mid = Math.abs(this.menuWidth + menu.left  - e.clientX)
  let midT = menu.height + menu.top > e.clientY && e.clientY > menu.top
  let event = e || window.event; 
  if (mid <= 20 && midT && event.ctrlKey){
    this.cursor = 'col-resize' 
    this.menuWidth = parseInt(e.clientX - menu.left) 
  }else{
    this.cursor = 'auto'
  }

}
```
