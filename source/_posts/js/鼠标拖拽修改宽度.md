<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>鼠标移动更改内容宽度</title>
  <style>
    *{
      padding:0;
      margin: 0;
    }
    .content{
      width: 100%;
      display: flex;
      cursor: pointer;
    }
    .content .menu{
      width: 750px;
      background-color: bisque;
      height: 400px;
    }
    .content .cont{
      flex: 1;
      background-color: aqua;
    }
  </style>
</head>
<body>
  鼠标移动更改内容宽度
 <div class="content" id="drags" >
    <div class="menu" id="menu">
    </div>
    <div class="cont">
    </div>
</div>


  <script>

    let drags = document.getElementById('drags');   
    let menuDom = document.getElementById('menu');  
    drags.addEventListener('mousemove',handleMove)
    function handleMove(e){
      let menu = menuDom.getBoundingClientRect()
      let mid = Math.abs(menu.width + menu.left  - e.clientX)
      let midT = menu.height + menu.top > e.clientY && e.clientY > menu.top
      let event = e || window.event; 
      if (mid <= 20 && midT && event.ctrlKey){
        drags.setAttribute("style","cursor:col-resize");
        menuDom.setAttribute("style",`width:${parseInt(e.clientX - menu.left) }px`)
      }else{
        drags.setAttribute("style","cursor:auto");
      }
    }


  </script>
</body>
</html>
